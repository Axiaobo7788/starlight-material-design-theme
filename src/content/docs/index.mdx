import { Card, CardGrid } from '@astrojs/starlight/components';

## Theme Surface

<CardGrid>
	<Card title="Plugin First" icon="puzzle">
		Install as a Starlight plugin and let the theme append its own CSS entrypoint.
	</Card>
	<Card title="Token Bridge" icon="code-branch">
		Map Material Design 3 `--md-sys-*` roles into Starlight's existing `--sl-*` contract.
	</Card>
	<Card title="Docs Density" icon="list-format">
		Keep navigation, prose, code, tables, and asides compact enough for wiki-style reference.
	</Card>
	<Card title="No Heavy Runtime" icon="server">
		Use Material Web as a visual and token reference, not as a required component runtime.
	</Card>
</CardGrid>
